<script>

    const reviews = [];


    document.getElementById('search-form').addEventListener('submit', function(event) {

        event.preventDefault();

        const searchAddress = document.getElementById('search-address').value.trim();

        const filteredReviews = reviews.filter(review => review.address.toLowerCase() === searchAddress.toLowerCase());

        displayReviews(filteredReviews);

    });


    document.getElementById('review-form').addEventListener('submit', function(event) {

        event.preventDefault();

        const review = {

            country: document.getElementById('country').value,

            address: document.getElementById('address').value.trim(),

            address2: document.getElementById('address2').value.trim(),

            city: document.getElementById('city').value.trim(),

            state: document.getElementById('state').value.trim(),

            zip: document.getElementById('zip').value.trim(),

            timeLived: document.getElementById('time-lived').value,

            rent: document.getElementById('rent').value,

            buildingRating: document.getElementById('building-rating').value,

            love: document.getElementById('love').value.trim(),

            hate: document.getElementById('hate').value.trim(),

            issues: document.getElementById('issues').value.trim(),

            wishKnew: document.getElementById('wish-knew').value.trim(),

            landlordName: document.getElementById('landlord-name').value.trim(),

            landlordRating: document.getElementById('landlord-rating').value,

            landlordDetail: document.getElementById('landlord-detail').value.trim(),

            additionalInfo: document.getElementById('additional-info').value.trim()

        };

        reviews.push(review);

        displayReviews([review]);

        document.getElementById('review-form').reset();

    });


    function displayReviews(reviewsToDisplay) {

        const reviewsSection = document.getElementById('reviews-section');

        reviewsSection.innerHTML = '';

        if (reviewsToDisplay.length === 0) {

            reviewsSection.innerHTML = '<p class="no-results">No results found.</p>';

        } else {

            reviewsToDisplay.forEach(review => {

                const reviewElement = document.createElement('div');

                reviewElement.classList.add('review-item');

                reviewElement.innerHTML = `

                    <p><strong>Address:</strong> ${review.address}</p>

                    <p><strong>City:</strong> ${review.city}</p>

                    <p><strong>State:</strong> ${review.state}</p>

                    <p><strong>ZIP:</strong> ${review.zip}</p>

                    <p><strong>Time Lived:</strong> ${review.timeLived}</p>

                    <p><strong>Rent:</strong> $${review.rent}</p>

                    <p><strong>Building Rating:</strong> ${review.buildingRating}/10</p>

                    <p><strong>Love:</strong> ${review.love}</p>

                    <p><strong>Hate:</strong> ${review.hate}</p>

                    <p><strong>Issues:</strong> ${review.issues}</p>

                    <p><strong>Wish Knew:</strong> ${review.wishKnew}</p>

                    <p><strong>Landlord:</strong> ${review.landlordName}</p>

                    <p><strong>Landlord Rating:</strong> ${review.landlordRating}/10</p>

                    <p><strong>Landlord Detail:</strong> ${review.landlordDetail}</p>

                    <p><strong>Additional Info:</strong> ${review.additionalInfo}</p>

                `;

                reviewsSection.appendChild(reviewElement);

            });

        }

    }

</script>
